Calcolo Fattoriale

Scrivi un programma in C++ che chieda all'utente di inserire un numero intero non negativo e calcoli il fattoriale di quel numero. 


#include <iostream>
using namespace std;

int main() {
    int n;
    long fattoriale = 1;
    
    cout << "Inserisci un numero intero: ";
    cin >> n;

    if (n < 0) {
        cout << "Errore: il fattoriale di un numero negativo non esiste." << endl;
    } else {
        for(int i = 1; i <= n; ++i) {
            fattoriale *= i;
        }
        cout << "Fattoriale di " << n << " = " << fattoriale << endl;
    }

    return 0;
}

Calcolare la superficie di due figure

Scrivere un programma che riceve in input 4 numeri: base e altezza di un triangolo e base e altezza di un rettangolo. In output stampa la superficie della figura più estesa; se le due figure hanno la stessa superficie, stampare "Le due figure hanno la stessa superficie." 


            
#include <iostream>
using namespace std;

int main() {
  setlocale(LC_ALL, "Italian");

  int baseT = 0, altezzaT = 0, baseR = 0, altezzaR = 0, areaR = 0;
  float areaT = 0; // dividendo per due può avere la virgola

  // richiedo i dati
  cout << "Inserisci la lunghezza della base del triangolo: " << endl;
  cin >> baseT;
  cout << "Inserisci la lunghezza dell'altezza del triangolo: " << endl;
  cin >> altezzaT;
  cout << "Inserisci la lunghezza della base del rettangolo: " << endl;
  cin >> baseR;
  cout << "Inserisci la lunghezza dell'altezza del rettangolo: " << endl;
  cin >> altezzaR;

  // calcolo le aree
  areaT = (float)baseT * (float)altezzaT / 2; // faccio un cast perché divido
  areaR = baseR * altezzaR;
  // stampo
  cout << "Il triangolo ha superficie " << areaT
       << "; il rettangolo ha superficie " << areaR << "." << endl;

  if(areaR>areaT){
    cout<<"Il rettangolo ha superficie maggiore.";
  } else if (areaT>areaR)
  {
    cout <<"Il triangolo ha superficie maggiore.";
  } else cout << "Le due figure hanno la stessa superficie.";
  return 0;
}

Matrice e diagonali

Creare una matrice 10*10 da riempire con i numeri da 0 a 99, e stamparla con delle tablature(\t) e andando correttamente a capo. Dopodiché stampare le due diagonali. 


    
#include <iostream>
using namespace std;

int main() {

  int matrice[10][10];
  int a = 0;
  // Riempio la matrice
  for (int i = 0; i < 10; i++) {

    for (int j = 0; j < 10; j++) {

      matrice[i][j] = a;
      a++;
    }
  }

  // Stampo la matrice
  for (int i = 0; i < 10; i++) {

    for (int j = 0; j < 10; j++) {

      cout << matrice[i][j];
      cout << "\t";
    }
    cout << "\n\n";
  }

  cout << "\n\n";

  // Stampo la diagonale 0,11,22,33...
  cout << "Prima diagonale: " << endl;
  for (int i = 0; i < 10; i++) {

    cout << matrice[i][i] << "\t";
  }

  cout << endl;

  // Stampo l'altra diagonale
  cout << "Seconda diagonale: " << endl;
  for (int i = 0; i < 10; i++) {

    cout << matrice[i][9 - i] << "\t";
  }
}
